EJEMPLO 6.2

Realizar la serie de Fourier Trigonometrica compacta y exponencial compleja para el ejemplo 6.2 del libro del Lathi

Exponencial compleja

d0=0;
dn=@(n) ((-8*sin((n*pi)/2)/(n^2*pi^2))/2)*j;
t0=-1/2;
tf=3/2;
f=@(t) (2*t).*(t>=-1/2 & t<1/2) + (2*(1-t).*(t>=1/2 & t<=3/2)) ;
armo=15;
a=-4;
b=4;
sfc(t0,tf,dn,d0,f,armo,a,b);

Trigonometrica compacta

c0=0;
cn=@(n) 8*sin((n*pi)/2)/(n^2*pi^2);
f=@(t) (2*t).*(t>=-1/2 & t<1/2) + (2*(1-t).*(t>=1/2 & t<=3/2)) ;
a=-4;
b=4;
t0=-1/2;
tf=3/2;
teta=-pi/2;
armo=15;
sft(t0,tf,cn,c0,teta,f,armo,a,b);

EJEMPLO 6.4

Realizar la serie de Fourier Trigonometrica compacta y exponencial compleja para el ejemplo 6.4 del libro del Lathi

Exponencial compleja

clear all
dn=@(n)(5734161139222659*sinh((pi*j*n)/2))/(18014398509481984*j*n);
d0=1/2;
armo=15;
f=@(t) 0.*(t>=-pi & t<=-pi/2) + 1.*(t>-pi/2 & t<=pi/2) + 0.*(t>pi/2 & t<=pi);
t0=-pi;
tf=pi;
a=-3*pi;
b=7*pi;
sfc(t0,tf,dn,d0,f,armo,a,b);

Trigonometrica compacta

clear all
c0=1/2;
cn=@(n) (2/(pi.*n)).*sin(n*pi/2);
f=@(t) 0.*(t>=-pi & t<-pi/2) + 1.*(t>=-pi/2 & t<=pi/2) + 0.*(t>pi/2 & t<=pi);
a=-3*pi;
b=7*pi;
teta=0;
armo=15;
t0=-pi;
tf=pi;
sft(t0,tf,cn,c0,teta,f,armo,a,b)

EJEMPLO PR08-1

Realizar la serie de Fourier Trigonometrica compacta y exponencial compleja para el ejemplo PR 08-1

Exponencial compleja

clear all
d0=0;
dn=@(n)-(exp(pi*j*n)*(pi*j*n - 1))/(2*j^2*n^2*pi^2) - (exp(-pi*j*n)*(pi*j*n + 1))/(2*j^2*n^2*pi^2);
armo=15;
f=@(t)t;
t0=-1;
tf=1;
a=-5;
b=5;
sfc(t0,tf,dn,d0,f,armo,a,b);

Trigonometrica compacta

clear all
f=@(t) t;
c0=0;
t0=-1;
tf=1;
a=-5;
b=5;
teta=-pi/2;
armo=15;
cn=@(n) 2.*((-pi.*n.*cos(pi.*n)+sin(pi.*n))/(pi.^2.*n.^2));
sft(t0,tf,cn,c0,teta,f,armo,a,b)

EJEMPLO PR08-2

Realizar la serie de Fourier Trigonometrica compacta y exponencial compleja para el ejemplo PR 08-2

Exponencial compleja

clear all
d0=4/3;
dn=@(n)(4/(pi^3*n^3)).*(2*pi*n*cos(pi*n));
armo=15;
f=@(t)t.^2;
t0=-2;
tf=2;
a=-5;
b=5;
sfc(t0,tf,dn,d0,f,armo,a,b);

Trigonometrica compacta

clear all
f=@(t) t.^2;
c0=4/3;
cn=@(n) (16/(pi^3*n^3))*(pi*n*cos(pi*n));
t0=-2;
tf=2;
a=-5;
b=5;
teta=0;
armo=15;
sft(t0,tf,cn,c0,teta,f,armo,a,b)
